#+TITLE: Emacs Cheat Sheet
#+AUTHOR: Dmitry
#+STARTUP: showeverything

* Базовые концепции
- =C-= :: Клавиша Control
- =M-= :: Клавиша Meta (обычно Alt)
- =S-= :: Клавиша Shift
- =RET= :: Клавиша Enter
- =C-u <число> <команда>= :: Выполнить команду указанное число раз (Универсальный аргумент)
- =M-!= :: Выполнить консольную команду (shell-command)

* Спасение и выход (Паника)
- =C-g= :: Отменить команду / Выйти из минибуфера (Кнопка паники)
- =C-x C-c= :: Выйти из Emacs
- =C-/= :: Отменить последнее действие (Undo)

* Справка и Поиск команд
- =C-h v= :: Посмотреть значение и описание переменной (describe-variable)
- =C-h f= :: Посмотреть описание функции (describe-function)
- =C-h k= :: Узнать, какая функция привязана к комбинации клавиш (describe-key)
- =C-h a= :: Поиск команд по ключевому слову (apropos-command). "Гугл по Emacs".
- =M-x find-library= :: Открыть исходный код установленного пакета (Elisp-файла).
- =M-x apropos= :: Глобальный поиск вообще по всему (команды, переменные, стили).

* Навигация и Поиск
- =C-n= :: Вниз на одну строку (Next line)
- =C-p= :: Вверх на одну строку (Previous line)
- =C-f= :: Вперед на один символ (Forward char)
- =C-b= :: Назад на один символ (Backward char)
- =M-f= :: Вперед на одно слово (Forward word)
- =M-b= :: Назад на одно слово (Backward word)
- =C-a= :: Перейти в начало строки (move-beginning-of-line)
- =C-e= :: Перейти в конец строки (move-end-of-line)
- =M-b= :: Назад на одно слово (Backward word)
- =M->= :: Перейти в конец буфера (Физически: Meta + Shift + точка)
- =M-<= :: Перейти в начало буфера (Физически: Meta + Shift + запятая)
- =C-s= :: Инкрементальный поиск вперед, повторно - переход к следующему
- =C-r= :: Инкрементальный поиск назад, повторно - переход к предыдущему
- =M-m= :: Перейти к первому непробельному символу в строке (back-to-indentation)
- =M-^= :: Присоединить текущую строку к концу предыдущей (delete-indentation / "join lines")
- =C-v= :: Прокрутить экран на страницу вниз (Page Down)
- =M-v= :: Прокрутить экран на страницу вверх (Page Up)  

* Редактирование и Буфер обмена (Kill Ring)
- =C-d= :: Удалить один символ вперед (не сохраняет в Kill Ring)
- =M-d= :: Вырезать слово вперед (сохраняет в Kill Ring)
- =C-k= :: Вырезать текст от курсора до конца строки
- =C-SPC= :: Поставить маркер для начала выделения (set-mark-command)
- =C-x h= :: Выделить весь буфер целиком (mark-whole-buffer)
- =C-w= :: Вырезать/Удалить выделенный текст (kill-region)
- =M-w= :: Скопировать выделенный текст
- =C-y= :: Вставить текст (yank)
- =M-y= :: Умная вставка из истории буфера обмена (consult-yank-pop / yank-pop)
- =M-%= :: Интерактивная замена текста (query-replace). После запуска нажать ! для замены всех совпадений.

* Файлы и Буферы
- =C-x C-f= :: Открыть или создать файл на диске (find-file)
- =C-x C-s= :: Сохранить текущий файл (save-buffer)
- =C-x d= :: Открыть встроенный файловый менеджер Dired
- =C-x b= :: Переключиться на другой буфер или создать новый (switch-to-buffer)
- =C-x k= :: Закрыть (убить) буфер (kill-buffer)

* Окна и Вкладки (Разметка экрана)
- =C-x 1= :: Оставить только текущее окно (Спрятать остальные)
- =C-x 2= :: Разделить окно по горизонтали (Одно над другим)
- =C-x 3= :: Разделить окно по вертикали (Одно рядом с другим)
- =C-x o= :: Переместить курсор в другое окно (other-window)
- =M-o= :: Быстрый переход в любое окно (ace-window)
- =C-x 0= :: Закрыть текущее окно (delete-window). Буфер при этом не удаляется!
- =q= :: Закрыть служебные буферы (справку, отладчик и т.д.)
- =C-x t 2= :: Создать новую вкладку (workspace)
- =C-x t 1= :: Закрыть все вкладки кроме текущей
- =C-x t 0= :: Закрыть текущую вкладку
- =C-x ]= :: Увеличить высоту текущего окна (enlarge-window)
- =C-x [= :: Уменьшить высоту текущего окна (shrink-window)
- =C-x }= :: Увеличить ширину текущего окна (enlarge-window-horizontally)
- =C-x {= :: Уменьшить ширину текущего окна (shrink-window-horizontally)
- =C-x += :: Сбалансировать размеры всех окон (сделать одинаковыми)

* Org-mode
- =Tab= :: Развернуть / Свернуть заголовок
- =S-Tab= :: Развернуть / Свернуть все заголовки (Shift + Tab)
- =S-Стрелки= :: Переключить статус TODO/DONE у заголовка
- =C-c C-c= :: Универсальное действие (чекбокс [X], выровнять таблицу и т.д.)

* Dired (Файловый менеджер)
- =n= :: Вниз на одну строку (next)
- =p= :: Вверх на одну строку (previous)
- =RET= :: Открыть файл или войти в директорию (Enter)
- =^= :: Подняться на уровень выше (родительская директория)
- =q= :: Скрыть буфер Dired (quit)
- =+= :: Создать новую директорию
- =C= :: Скопировать файл или отмеченные файлы (Copy)
- =R= :: Переименовать или переместить файл(ы) (Rename)
- =D= :: Удалить файл(ы) (Delete)
- =m= :: Отметить файл для групповой операции (mark)
- =u= :: Снять отметку (unmark)
- =U= :: Снять все отметки в буфере (Unmark all)
- =g= :: Обновить содержимое буфера (refresh), если файлы изменились извне

* Magit

- =C-x g= :: Открыть панель статуса Magit (magit-status).
- =s= :: (В панели Magit) Stage - добавить файл или изменение в индекс (git add).
- =c c= :: (В панели Magit) Открыть окно создания коммита (git commit).
- =C-c C-c= :: Сохранить сообщение и завершить коммит.
- =g= :: (В панели Magit) Обновить статус (refresh)
- =q= :: Закрыть панель/окно Magit.
- =M= :: (В панели Magit) Управление удаленными репозиториями (Remotes).
- =P= :: (В панели Magit) Отправка коммитов на сервер (Push)
- =i= :: (В панели Magit) Быстро добавить файл или папку в .gitignore.

* Шпаргалка по префиксам коммитов (Conventional Commits)
- =feat:= :: Добавление нового функционала или пакета. Пример: =feat(ui): добавлена панель вкладок=
- =fix:= :: Исправление ошибки. Пример: =fix(fonts): исправлен размер шрифта=
- =chore:= :: Рутина, обслуживание, игнорирование файлов. Пример: =chore: добавлен transient в .gitignore=
- =refactor:= :: Улучшение структуры кода без изменения его работы. Пример: =refactor(init): код разделен на блоки=
- =docs:= :: Написание комментариев или документации. Пример: =docs: добавлены комментарии к use-package=
- =style:= :: Изменения форматирования (пробелы, переносы и т.д.), не влияющие на логику.
- =perf:= :: Изменения, направленные на улучшение производительности.
- =test:= :: Добавление или исправление тестов.
